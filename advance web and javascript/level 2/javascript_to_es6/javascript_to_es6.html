Logic for bubble sort ascending
function bubble_sort_asc(a)
{
 let swapp;
 let n = a.length-1;
 const x = a;

 do
 {
  swapp = false;

  for(let i=0; i<\n; i++)
  {
   if(x[i] > x[i+1])
   {
    let temp = x[i];
    x[i] = x[i+1];
    x[i+1] = temp;
    swapp = true;
   }
  }
  n--;
 } while (swapp);

return x;
}
